{
    \pgfmathsetmacro{\n}{#5}
    \draw[#1] (0,0) rectangle (#2*\n,#2);
    \foreach \b in {1,...,\n}{
        \draw[#1,-] (#2*\b,0) to (#2*\b,#2);
        \pgfmathtruncatemacro{\A}{\n-\b}
        \node at (#2*\b-#2/2,-0.25) {$#3^{\A}$};
        \node at (#2*\b-#2/2,0.35) {$\StrChar{#4}{\b}$};
    }
}
