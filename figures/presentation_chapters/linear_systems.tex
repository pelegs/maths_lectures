\newcommand{\form}{
  $\left\{
  \begin{array}{ccccc}
	\tikzmarkin[set fill color=col1!50, set border color=col1]{A}2x & - & y & = & 1\tikzmarkend{A}\\
	\tikzmarkin[set fill color=col2!50, set border color=col2]{B}x & + & y & = & 2\tikzmarkend{B}\\
  \end{array}
  \right.
  \Rightarrow \begin{pmatrix} 2 & -1 \\ 1 & 1 \end{pmatrix}\colvec{2}{x}{y} = \colvec{2}{1}{2}$
}
